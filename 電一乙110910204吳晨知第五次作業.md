### d
程式碼

```c
#include <stdio.h>
#define SIZE 4
#define SIZE_2 5


int main(void) {
  int n[SIZE][SIZE_2] = {
    {64, 43, 86, 22, 76},
    {47, 4, 54, 94, 99},
    {80, 0, 72, 10, 87},
    {74, 55, -8, 50, 63},
  };
  int max = n[0][0];
  int min = n[0][0];

  for (size_t i = 0; i < SIZE; ++i) {
    for (size_t j = 0; j < SIZE_2; ++j) {
      if (max < n[i][j]) {
        max = n[i][j];
      } else if (min > n[i][j]) {
        min = n[i][j];
      }
      printf("%3d ", n[i][j]);
    }
    printf("\n");
  }
  puts("");
  printf("%10s%10s\n", "Max", "Min");
  printf("%10d%10d\n", max, min);
}
```

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 100


int main(void) {
  int d1[SIZE];
  int d2[SIZE];
  int sum[SIZE];
  int diff[SIZE];
  srand(time(NULL));

  for (size_t i = 0; i < 2; ++i) {
    for (size_t j = 0; j < SIZE; ++j) {
      switch (i) {
        case 0:
          d1[j] = rand() % 100;
          printf("%4d", d1[j]);
          break;
        case 1:
          d2[j] = rand() % 100;
          printf("%4d", d2[j]);
          break;
      }
      if (((j + 1) % 10) == 0) {
        printf("\n");
      }
    }
    puts("");
  }


  for (size_t i = 0; i < 2; ++i) {
    for (size_t j = 0; j < SIZE; ++j) {
      switch (i) {
        case 0:
          sum[j] = d1[j] + d2[j];
          printf("%4d", sum[j]);
          break;
        case 1:
          diff[j] = d1[j] - d2[j];
          printf("%4d", diff[j]);
          break;
      }
      if (((j + 1) % 10) == 0) {
        printf("\n");
      }
    }
    puts("");
  }
}
