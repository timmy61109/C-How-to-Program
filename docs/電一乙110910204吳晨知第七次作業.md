};

void deal(const Card * const wDeck);

  Card deck[CARDS];
  srand(time(NULL));
  fillDeck(deck);
  shuffle(deck);
  deal(deck);

void fillDeck(Card * const wDeck) {
  for (size_t i = 0; i < CARDS; ++i) {
    wDeck[i].face = i % (CARDS / 4);
    wDeck[i].suit = i / (CARDS / 4);
  }
}

void shuffle(Card * const wDeck) {
  for (size_t i = 0; i < CARDS; ++i) {
    size_t j = rand() % CARDS;
    Card temp = wDeck[i];
    wDeck[i] = wDeck[j];
    wDeck[j] = temp;
  }
}

  char face[13][10] = {
